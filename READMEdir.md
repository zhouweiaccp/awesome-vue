## es6开发环境
npm install babel-cli babel-eslint -g
babel --version
mkdir es6
cd es6
npm init
npm install --save-dev babel-preset-es2015 babel-cli   #npm install --save-dev babel-preset-eslatest-node6
babel lib -d build\lib
 babel src/index.js -o dist/index.js
 npm run build
